"""

Description: Jeu interactif avec interface graphique Tkinter
"""

import tkinter as tk
from tkinter import messagebox
import random


class ChasseTresor:
    def __init__(self, root):
        self.root = root
        self.root.title("üéÆ Chasse au Tr√©sor")
        self.root.geometry("600x700")
        self.root.configure(bg="#2c3e50")
        
        # Variables du jeu
        self.taille_grille = 5
        self.score = 0
        self.tentatives = 0
        self.max_tentatives = 15
        
        # Positions des tr√©sors et pi√®ges
        self.positions_tresors = self.generer_positions(3)
        self.positions_pieges = self.generer_positions(2)
        self.cases_revelees = set()
        
        self.creer_interface()
    
    def generer_positions(self, nombre):
        """G√©n√®re des positions al√©atoires uniques"""
        positions = set()
        while len(positions) < nombre:
            x = random.randint(0, self.taille_grille - 1)
            y = random.randint(0, self.taille_grille - 1)
            positions.add((x, y))
        return positions
    
    def creer_interface(self):
        """Cr√©e l'interface graphique"""
        # Titre
        titre = tk.Label(
            self.root,
            text="üó∫Ô∏è CHASSE AU TR√âSOR üíé",
            font=("Arial", 24, "bold"),
            bg="#2c3e50",
            fg="#ecf0f1",
            pady=20
        )
        titre.pack()
        
        # Frame pour les informations
        info_frame = tk.Frame(self.root, bg="#2c3e50")
        info_frame.pack(pady=10)
        
        self.label_score = tk.Label(
            info_frame,
            text=f"üí∞ Score: {self.score}",
            font=("Arial", 16, "bold"),
            bg="#27ae60",
            fg="white",
            padx=20,
            pady=10,
            borderwidth=3,
            relief="raised"
        )
        self.label_score.pack(side=tk.LEFT, padx=10)
        
        self.label_tentatives = tk.Label(
            info_frame,
            text=f"üéØ Tentatives: {self.tentatives}/{self.max_tentatives}",
            font=("Arial", 16, "bold"),
            bg="#3498db",
            fg="white",
            padx=20,
            pady=10,
            borderwidth=3,
            relief="raised"
        )
        self.label_tentatives.pack(side=tk.LEFT, padx=10)
        
        # Instructions
        instructions = tk.Label(
            self.root,
            text="Trouvez les 3 tr√©sors cach√©s ! Attention aux pi√®ges !",
            font=("Arial", 12),
            bg="#2c3e50",
            fg="#ecf0f1",
            pady=10
        )
        instructions.pack()
        
        # Frame pour la grille
        grille_frame = tk.Frame(self.root, bg="#2c3e50")
        grille_frame.pack(pady=20)
        
        # Cr√©er la grille de boutons
        self.boutons = []
        for i in range(self.taille_grille):
            ligne = []
            for j in range(self.taille_grille):
                btn = tk.Button(
                    grille_frame,
                    text="‚ùì",
                    font=("Arial", 20, "bold"),
                    width=4,
                    height=2,
                    bg="#34495e",
                    fg="white",
                    activebackground="#7f8c8d",
                    command=lambda x=i, y=j: self.cliquer_case(x, y)
                )
                btn.grid(row=i, column=j, padx=5, pady=5)
                ligne.append(btn)
            self.boutons.append(ligne)
        
        # Bouton rejouer
        btn_rejouer = tk.Button(
            self.root,
            text="üîÑ Nouvelle Partie",
            font=("Arial", 14, "bold"),
            bg="#e74c3c",
            fg="white",
            padx=20,
            pady=10,
            command=self.rejouer,
            cursor="hand2"
        )
        btn_rejouer.pack(pady=20)
        
        # L√©gende
        legende = tk.Label(
            self.root,
            text="üíé Tr√©sor (+10 pts) | üí£ Pi√®ge (-5 pts) | ‚¨ú Vide (0 pt)",
            font=("Arial", 10),
            bg="#2c3e50",
            fg="#95a5a6",
            pady=10
        )
        legende.pack()
    
    def cliquer_case(self, x, y):
        """G√®re le clic sur une case"""
        if (x, y) in self.cases_revelees:
            return
        
        if self.tentatives >= self.max_tentatives:
            messagebox.showinfo("Fin du jeu", "Plus de tentatives !")
            return
        
        self.tentatives += 1
        self.cases_revelees.add((x, y))
        btn = self.boutons[x][y]
        
        # V√©rifier si c'est un tr√©sor
        if (x, y) in self.positions_tresors:
            btn.config(text="üíé", bg="#f39c12", state="disabled")
            self.score += 10
            self.positions_tresors.remove((x, y))
            
            if len(self.positions_tresors) == 0:
                messagebox.showinfo(
                    "üéâ Victoire !",
                    f"Bravo ! Vous avez trouv√© tous les tr√©sors !\nScore final: {self.score} points"
                )
        
        # V√©rifier si c'est un pi√®ge
        elif (x, y) in self.positions_pieges:
            btn.config(text="üí£", bg="#e74c3c", state="disabled")
            self.score = max(0, self.score - 5)
            self.positions_pieges.remove((x, y))
        
        # Case vide
        else:
            btn.config(text="‚¨ú", bg="#95a5a6", state="disabled")
        
        # Mettre √† jour l'affichage
        self.mettre_a_jour_affichage()
        
        # V√©rifier la fin du jeu
        if self.tentatives >= self.max_tentatives:
            messagebox.showinfo(
                "Fin du jeu",
                f"Plus de tentatives !\nScore final: {self.score} points"
            )
    
    def mettre_a_jour_affichage(self):
        """Met √† jour le score et les tentatives"""
        self.label_score.config(text=f"üí∞ Score: {self.score}")
        self.label_tentatives.config(
            text=f"üéØ Tentatives: {self.tentatives}/{self.max_tentatives}"
        )
    
    def rejouer(self):
        """Recommence une nouvelle partie"""
        self.score = 0
        self.tentatives = 0
        self.positions_tresors = self.generer_positions(3)
        self.positions_pieges = self.generer_positions(2)
        self.cases_revelees = set()
        
        # R√©initialiser tous les boutons
        for i in range(self.taille_grille):
            for j in range(self.taille_grille):
                self.boutons[i][j].config(
                    text="‚ùì",
                    bg="#34495e",
                    state="normal"
                )
        
        self.mettre_a_jour_affichage()


if __name__ == "__main__":
    root = tk.Tk()
    jeu = ChasseTresor(root)
    root.mainloop()

üéØ R√®gles du Jeu

  • Tentatives : Vous avez 15 clics maximum

  • Pi√®ge üí£ : -5 points
